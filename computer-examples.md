# Примеры программ

## Demo
```asm
jmp start_title		; второй байт перезаписывается

"Привет,\nОнигири!\n"	; задом-наперёд

start_title:
  ldi b, 0x0A		; "\n"
  st b, 0x01

ldi b, end_of_title
ldi c, 0x3C		; terminal
ldi d, loop

loop:
  ld a, b
  st a, c
  dec b
  jnz d

hlt

; заполнение пустыми байтами

; подключение дисплея и терминала
0x05			; порт ввода/вывода

0x00

; картинка для дисплея в области 0x40...0x5F

```

## Terminal
Выводит в терминал текст, набираемый на клавиатуре.
```asm
inc a		; 0b00000001
ldi b, 0x3E	; порт ввода/вывода
st a, b		; подключение терминала
ldi c, 0x3C	; адрес ввода в терминал
ldi d, loop	; адрес для повторения цикла
loop:
  ld a, b	; чтение нажатой клавиши
  st a, c	; отправка в терминал
  jmp d		; повторение цикла
```
